<script>
  function m(value) {
    document.getElementById("report1").innerHTML = "";
    
        yellowfin.loadReport({
            reportId: '63027943-3be0-4812-9017-314f21cf0927',
            element: document.querySelector('#report1'),
            showToolbar: false,
            filterValues: [{
                filterId: 'Demographic',
                valueList: [value]
            }]     
        }).then(report => {
            console.log(report);
        });
    
};

};

a();
